!! ! SA-R (St. Augustine)

config t

map-class frame-relay CIR_SA_CF
 frame-relay cir 64000
 exit

map-class frame-relay CIR_SA_STUD1
 frame-relay cir 256000
 exit

interface serial0/0
 description SA-R to FR3 link
 no ip address
 encapsulation frame-relay
 frame-relay traffic-shaping
 shutdown
 no shutdown
 exit

interface serial0/0.210 point-to-point
 description Link to CF-R
 ip address 10.0.0.6 255.255.255.252
 frame-relay interface-dlci 210
 class CIR_SA_CF
 exit
 exit

interface serial0/0.305 point-to-point
 description Link to FRX1
 ip address 10.0.0.9 255.255.255.252
 frame-relay interface-dlci 305
 class CIR_SA_STUD1
 exit
 exit
