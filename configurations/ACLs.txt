!! ! STUD1

config t

ip access-list extended ACL_STUD1

! HTTP
permit tcp 172.16.30.0 0.0.0.31 172.16.60.0 0.0.0.31 eq 80

! HTTPS
permit tcp 172.16.30.0 0.0.0.31 172.16.60.0 0.0.0.31 eq 443

! eBGP
permit tcp 172.16.30.0 0.0.0.31 172.16.60.0 0.0.0.31 eq 179

! ICMP
permit icmp 172.16.30.0 0.0.0.31 172.16.60.0 0.0.0.31

! DHCP
permit udp 172.16.30.0 0.0.0.31 172.16.60.0 0.0.0.31 eq 67
permit udp 172.16.30.0 0.0.0.31 172.16.60.0 0.0.0.31 eq 68

! NTP
permit udp 172.16.30.0 0.0.0.31 172.16.60.0 0.0.0.31 eq 123

! DNS
permit udp 172.16.30.0 0.0.0.31 172.16.60.0 0.0.0.31 eq 53
permit tcp 172.16.30.0 0.0.0.31 172.16.60.0 0.0.0.31 eq 53

! SSH
permit tcp 172.16.30.0 0.0.0.31 172.16.60.0 0.0.0.31 eq 22

! Deny any
deny ip any any

interface fa0/0
 ip access-group ACL_STUD1 in
 end

show ip access-lists
